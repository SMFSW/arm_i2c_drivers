FctERR $CPNT$_fct(void)
{
	FctERR err;

	err = $CPNT$_Read($CPNT$.$CPNT$_val_1, $CPNT$__REG1, 1);
	if (err)	{ return err; }

	return err;
}


/****************************************************************/


__WEAK FctERR NONNULL__ $CPNT$_INT_GPIO_Init(GPIO_TypeDef * const GPIOx, const uint16_t GPIO_Pin, const GPIO_PinState GPIO_Active) {
	return I2C_peripheral_GPIO_init(&$CPNT$.cfg.INT_GPIO, GPIOx, GPIO_Pin, GPIO_Active); }

__WEAK FctERR NONNULL__ $CPNT$_INT_GPIO_Get(bool * const pState) {
	return I2C_peripheral_GPIO_get(&$CPNT$.cfg.INT_GPIO, pState); }


__WEAK FctERR NONNULL__ $CPNT$_RST_GPIO_Init(GPIO_TypeDef * const GPIOx, const uint16_t GPIO_Pin, const GPIO_PinState GPIO_Active) {
	return I2C_peripheral_GPIO_init(&$CPNT$.cfg.RST_GPIO, GPIOx, GPIO_Pin, GPIO_Active); }

__WEAK FctERR $CPNT$_RST_GPIO_Set(const bool state) {
	return I2C_peripheral_GPIO_set(&$CPNT$.cfg.RST_GPIO, state); }
